let gradeData = [] let passfailData = [] async function loadData() { const gradeRes = await fetch('data/grade_students.json') gradeData = await gradeRes.json() const passRes = await fetch('data/passfail_students.json') passfailData = await passRes.json() } // โหลดตอนเปิดเว็บ loadData() function searchStudent() { const id = document.getElementById("searchBox").value.trim() const grades = gradeData.filter(s => s.id == id) const passfails = passfailData.filter(s => s.id == id) const result = document.getElementById("result") if (grades.length === 0 && passfails.length === 0) { result.innerHTML = "ไม่พบข้อมูล" return } // 🧑‍🎓 ข้อมูลนักเรียน const student = grades[0] || passfails[0] let html = `

${student.name}

ชั้น ${student.year}

` // ==================== // 📊 ตารางเกรด // ==================== if (grades.length > 0) { html += ` ` grades.forEach(g => { html += ` ` }) html += `
รหัสวิชา ชื่อวิชา เกรด
${g.subject_code} ${g.subject_name} ${g.grade}
` } // ==================== // ✅ ผ่าน / ไม่ผ่าน // ==================== if (passfails.length > 0) { html += `

ผลการประเมินเพิ่มเติม

` passfails.forEach(p => { html += `

${p.subject_name} : ${p.result}

` }) } result.innerHTML = html html += `

GPA: ${student.gpa}

`
